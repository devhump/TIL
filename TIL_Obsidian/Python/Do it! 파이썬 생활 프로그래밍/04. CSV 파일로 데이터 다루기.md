```python
def writecsv(filename, the_list):
# 첫번째 변수는 만들 파일 이름, 두번째는 CSV형 리스트를 저장한 객
```

```python
### 문자가 섞인 원소 골라내기
>>> k = []
>>> for j in i:
>>> 	if re.search('[a-z가-힣]', j): 
>>> 	# 알파벳 또는 한글이 있다면 그대로 저장

```

```python
import re

def swithch(listName):
	for i in listName:
		for j in i:
			try:
				i[i.index(j)] = float(re.sub(',', '', j))
			except:
				pass
	return listName
```
