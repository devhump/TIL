#### jQuery 에서 특정 클래스 삭제 / 추가 하는 방법
```js
// 클래스 삭제
$('.tab-button').removeClass('orange');
// 여러 클래스 한번에 삭제
$('.tab-button').removeClass('orange red purple');
  
// 클래스 추가
$('.tab-button').addClass('orange');
// 여러 클래스 한번에 추가
$('.tab-button').addClass('orange red purple');
```
