	graph[a].append((b, c))

def dijkstra(start):
	# μ‹μ‘ λ…Έλ“μ— λ€ν•΄μ„ μ΄κΈ°ν™”
	q = []
	# μ‹μ‘ λ…Έλ“λ΅ κ°€κΈ° μ„ν• μµλ‹¨ κ²½λ΅λ” 0μΌλ΅ μ„¤μ •ν•μ—¬, νμ— μ‚½μ…
	heapq.heappush(q, (0, start))
